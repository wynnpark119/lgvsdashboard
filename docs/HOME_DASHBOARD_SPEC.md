interface IntensityInputs {
  // LG.com
  pageViews: number;
  avgTimeOnPage: number;  // seconds
  revisitRate: number;    // %
  
  // YouTube
  videoViews: number;
  avgViewPercentage: number;  // %
  
  // LinkedIn
  engagements: number;
  
  // Engagement Signals
  webinarAttendance: number;
  inquiryCount: number;
  newsletterClicks: number;
}

function calculateIntensity(inputs: IntensityInputs): number {
  // ê°€ì¤‘ì¹˜
  const weights = {
    pageViews: 0.10,
    avgTimeOnPage: 0.15,
    revisitRate: 0.15,
    videoViews: 0.10,
    avgViewPercentage: 0.15,
    engagements: 0.10,
    webinarAttendance: 0.10,
    inquiryCount: 0.10,
    newsletterClicks: 0.05,
  };
  
  // ì •ê·œí™” (0-10 scale)
  const normalized = {
    pageViews: normalize(inputs.pageViews, 0, 10000),
    avgTimeOnPage: normalize(inputs.avgTimeOnPage, 0, 300),
    revisitRate: normalize(inputs.revisitRate, 0, 50),
    videoViews: normalize(inputs.videoViews, 0, 100000),
    avgViewPercentage: normalize(inputs.avgViewPercentage, 0, 100),
    engagements: normalize(inputs.engagements, 0, 5000),
    webinarAttendance: normalize(inputs.webinarAttendance, 0, 100),
    inquiryCount: normalize(inputs.inquiryCount, 0, 50),
    newsletterClicks: normalize(inputs.newsletterClicks, 0, 500),
  };
  
  // ê°€ì¤‘ í•©ì‚°
  let score = 0;
  for (const [key, weight] of Object.entries(weights)) {
    score += normalized[key] * weight * 10;
  }
  
  return Math.min(10, Math.max(1, score));
}

function normalize(value: number, min: number, max: number): number {
  return Math.min(1, Math.max(0, (value - min) / (max - min)));
}
